%Custom commands
\newcommand{\resheading}[3][3]{%
  \noindent
  \par\vspace{1.5em}
  \begin{tikzpicture}[baseline]
    % 아이콘 원
    \node[circle, fill=pricolor, minimum size=1.8em, inner sep=0pt] (icon) at (0, 0) 
      {\textcolor{white}{#3}};  % #3: 아이콘 (예: \faBook)
    
    % 텍스트 + 라인
    \node[anchor=west, inner sep=0pt, outer sep=0pt] (text) at ([xshift=1em]icon.east) {%
      \textbf{\Large%
        {\textcolor{pricolor}{\StrLeft{#2}{#1}}%
        \StrGobbleLeft{#2}{#1}}%
      }%
    };
    
    % 수평선
    \draw[backcolor, line width=1pt] 
      ([xshift=0.5em] text.east) -- (\linewidth, 0);
  \end{tikzpicture}
  \par\vspace{0.5em}
}
